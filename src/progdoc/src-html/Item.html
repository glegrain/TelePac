002/**
003 * Item class
004 * 
005 * @author Guillaume Legrain
006 * @version 2013-10-05
007 */
008public class Item
009{
010    // instance variables
011    private String aDescription;
012    private double aWeight;
013    
014
015    /**
016     * Constructor for objects of class Item
017     * 
018     * @param pDescription The item's description
019     * @param pWeight The item's weight
020     */
021    public Item(final String pDescription, final double pWeight)
022    {
023        // initialise instance variables
024       this.aDescription = pDescription;
025       this.aWeight = pWeight;
026    }
027    
028    /**
029     * Get Item's description
030     * @return The Item's description
031     */
032    public String getDescription()
033    {
034        return this.aDescription;
035    }
036    
037    @Override
038    public String toString()
039    {
040        return this.getDescription();
041    }
042    
043    /**
044     * Get Item's weight
045     * @return The Item's weight
046     */
047    public double getWeight()
048    {
049        return this.aWeight;
050    }
051    
052}
